import math

def solve_quadratic(a, b, c):
    """
    Solves a quadratic equation ax² + bx + c = 0
    Returns the roots or a message if no real solutions exist
    """
    
    # Calculate the discriminant
    discriminant = b**2 - 4*a*c
    
    if discriminant < 0:
        return "No real solutions (discriminant is negative)"
    elif discriminant == 0:
        # One solution
        x = -b / (2*a)
        return f"One solution: x = {x}"
    else:
        # Two solutions
        sqrt_discriminant = math.sqrt(discriminant)
        x1 = (-b + sqrt_discriminant) / (2*a)
        x2 = (-b - sqrt_discriminant) / (2*a)
        return f"Two solutions: x1 = {x1}, x2 = {x2}"

# Example usage
a = 1
b = -5
c = 6

result = solve_quadratic(a, b, c)
print(result)
